/**
 * @param {number[]} nums
 * @param {number} target
 * @return {number[]}
 */

var twoSum = function(nums, target) 
{
    for(let i = 0;i<nums.length;i++)
    {
        let temp = target - nums[i]
        for(let j = i+1;j<nums.length;j++)
        {
            if(temp == nums[j])
            return [i,j]
        }

    }
};


//两次循环遍历，依次判断每个数是不是target-nums[i]
