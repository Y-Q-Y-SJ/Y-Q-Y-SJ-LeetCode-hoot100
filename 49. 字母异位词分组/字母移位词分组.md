/**
 * @param {string[]} strs
 * @return {string[][]}
 */
var groupAnagrams = function(strs) {
    let hashmap = new Map()
    for(let str of strs)
    {
        let arry = Array.from(str)
        arry.sort()
        let key = arry.toString()
        let list = hashmap.get(key) ? hashmap.get(key) : new Array()
        list.push(str)
        hashmap.set(key, list)
    }
    return Array.from(hashmap.values());
};
